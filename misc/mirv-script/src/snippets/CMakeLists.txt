cmake_minimum_required (VERSION 3.24)

project ("AfxHookSource2Snippets")

set(AFXHOOKSOURCE2SNIPPETS_NAMES
    mirv_script_aim
    mirv_script_fov
    mirv_script_point
	mirv_script_spec_lock
    mirv_script_view
    mirv_script_voice
    tests/observer-test
    tests/attachment-test
	tests/cvars-test
    quit-when-demo-is-paused
)

list(TRANSFORM AFXHOOKSOURCE2SNIPPETS_NAMES APPEND ".ts" OUTPUT_VARIABLE AFXHOOKSOURCE2SNIPPETS_INPUTS)
list(TRANSFORM AFXHOOKSOURCE2SNIPPETS_INPUTS PREPEND "${AfxHookSource2Snippets_SOURCE_DIR}/" OUTPUT_VARIABLE AFXHOOKSOURCE2SNIPPETS_INPUTS)

set(AFXHOOKSOURCE2SNIPPETS_TSCONFIG "${AfxHookSource2Snippets_BINARY_DIR}/tsconfig.snippets.autogenerated.json")
set(AFXHOOKSOURCE2SNIPPETS_PROJECT_DIR "${AfxHookSource2Snippets_SOURCE_DIR}/../../")
file(GLOB AFXHOOKSOURCE2SNIPPETS_TYPE_INPUTS CONFIGURE_DEPENDS
    "${AFXHOOKSOURCE2SNIPPETS_PROJECT_DIR}/src/types/*.d.ts"
)
set(AFXHOOKSOURCE2SNIPPETS_TSC_FILES "")
foreach(AFXHOOKSOURCE2SNIPPETS_INPUT IN LISTS AFXHOOKSOURCE2SNIPPETS_TYPE_INPUTS)
    string(APPEND AFXHOOKSOURCE2SNIPPETS_TSC_FILES "\"${AFXHOOKSOURCE2SNIPPETS_INPUT}\",")
endforeach()
foreach(AFXHOOKSOURCE2SNIPPETS_INPUT IN LISTS AFXHOOKSOURCE2SNIPPETS_INPUTS)
    string(APPEND AFXHOOKSOURCE2SNIPPETS_TSC_FILES "\"${AFXHOOKSOURCE2SNIPPETS_INPUT}\",")
endforeach()
string(REGEX REPLACE ",$" "" AFXHOOKSOURCE2SNIPPETS_TSC_FILES "${AFXHOOKSOURCE2SNIPPETS_TSC_FILES}")

file(WRITE "${AFXHOOKSOURCE2SNIPPETS_TSCONFIG}" "{
	\"extends\": \"${AFXHOOKSOURCE2SNIPPETS_PROJECT_DIR}/tsconfig.json\",
	\"compilerOptions\": {
		\"rootDir\": \"${AfxHookSource2Snippets_SOURCE_DIR}\",
		\"outDir\": \"${AfxHookSource2Snippets_BINARY_DIR}/dist\",
		\"noEmit\": false,
	},
	\"include\": [],
	\"files\": [${AFXHOOKSOURCE2SNIPPETS_TSC_FILES}]
}")


list(TRANSFORM AFXHOOKSOURCE2SNIPPETS_NAMES APPEND ".js" OUTPUT_VARIABLE AFXHOOKSOURCE2SNIPPETS_OUTPUTS)
list(TRANSFORM AFXHOOKSOURCE2SNIPPETS_OUTPUTS PREPEND "${AfxHookSource2Snippets_BINARY_DIR}/dist/")

SET(AFXHOOKSOURCE2SNIPPETS_OUTPUTS ${AFXHOOKSOURCE2SNIPPETS_OUTPUTS} PARENT_SCOPE)

set(AFXHOOKSOURCE2SNIPPETS_NPM_CI_STAMP "${AfxHookSource2Snippets_BINARY_DIR}/npm-ci.stamp")

add_custom_command(
    OUTPUT "${AFXHOOKSOURCE2SNIPPETS_NPM_CI_STAMP}"
    DEPENDS
        "${AfxHookSource2Snippets_SOURCE_DIR}/../../package.json"
        "${AfxHookSource2Snippets_SOURCE_DIR}/../../package-lock.json"
    COMMAND npm ci
    COMMAND ${CMAKE_COMMAND} -E touch "${AFXHOOKSOURCE2SNIPPETS_NPM_CI_STAMP}"
    WORKING_DIRECTORY "${AfxHookSource2Snippets_SOURCE_DIR}/../../"
    VERBATIM
)

add_custom_target(AfxHookSource2SnippetsNpmCI
    DEPENDS "${AFXHOOKSOURCE2SNIPPETS_NPM_CI_STAMP}"
)

add_custom_command(
    OUTPUT ${AFXHOOKSOURCE2SNIPPETS_OUTPUTS}
    COMMAND ${CMAKE_COMMAND} -E rm -rf "${AfxHookSource2Snippets_BINARY_DIR}/dist"
    COMMAND npm run tsc -- --project "${AFXHOOKSOURCE2SNIPPETS_TSCONFIG}"
    DEPENDS "${AFXHOOKSOURCE2SNIPPETS_NPM_CI_STAMP}" "${AFXHOOKSOURCE2SNIPPETS_TSCONFIG}" ${AFXHOOKSOURCE2SNIPPETS_TYPE_INPUTS} ${AFXHOOKSOURCE2SNIPPETS_INPUTS}
    WORKING_DIRECTORY "${AfxHookSource2Snippets_SOURCE_DIR}/../../"
    VERBATIM
)

add_custom_target(AfxHookSource2Snippets
    DEPENDS ${AFXHOOKSOURCE2SNIPPETS_OUTPUTS}    
)
